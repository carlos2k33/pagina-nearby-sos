// src/app/robots.ts
import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      // Opcional: deshabilitar páginas que no quieres que Google indexe
      // disallow: ['/admin/', '/api/'],
    },
    sitemap: 'https://nearbysos.cl/sitemap.xml',
  }
}
